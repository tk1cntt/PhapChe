Add ReviewChecklistAnswer model after the existing Review model in prisma/schema.prisma:

```prisma
model Review {
  id          String       @id @default(cuid())
  workspaceId String
  requestId   String
  documentId  String
  reviewerId  String
  documentVersionId String  // NEW: link to specific document version
  status      ReviewStatus @default(in_progress)  // NEW: in_progress/approved/rejected
  decision    ReviewDecision?  // NEW: approve/reject
  generalComment String?  // NEW: general revision comment
  completedAt DateTime?  // NEW: when review was completed
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt
  workspace   Workspace    @relation(fields: [workspaceId], references: [id])
  request     LegalRequest @relation(fields: [requestId], references: [id])
  document    Document     @relation(fields: [documentId], references: [id])
  reviewer    User         @relation(fields: [reviewerId], references: [id])
  checklistAnswers ReviewChecklistAnswer[]  // NEW: link to checklist answers

  @@index([workspaceId])
  @@index([requestId])
  @@index([documentId])
  @@index([reviewerId])
  @@index([documentVersionId])  // NEW
  @@index([status])  // NEW
}

// NEW: Review status enum
enum ReviewStatus {
  in_progress
  approved
  rejected
}

// NEW: Review decision enum
enum ReviewDecision {
  approve
  reject
}

// NEW: Checklist answer model
model ReviewChecklistAnswer {
  id          String   @id @default(cuid())
  reviewId    String
  checklistItemId String  // References static checklist item ID (e.g., "formal-1", "legal-1")
  passed      Boolean  @default(false)
  comment     String?  // Optional per-item comment for failures
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  review      Review   @relation(fields: [reviewId], references: [id], onDelete: Cascade)

  @@unique([reviewId, checklistItemId])  // One answer per item per review
  @@index([reviewId])
}
```
</action>
