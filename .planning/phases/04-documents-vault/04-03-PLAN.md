    ```typescript
    function prefillVariables(schema: TemplateVariable[]) {
      const filled: Record<string, string> = {};
      for (const variable of schema) {
        const answerValue = initialAnswers[variable.key];
        if (answerValue !== undefined && answerValue !== null) {
          filled[variable.key] = String(answerValue);
        } else {
          filled[variable.key] = '';
        }
      }
      return filled;
    }
    ```

    **Variable Input Form:**

      - number type: <input type="number">

    - Label with required indicator (*)
    - Pre-filled value from variables state
    - On change: update variables state

    - Validate all required variables have non-empty values
    - If missing required: set error with Vietnamese message listing missing fields
    - Call server action: generateDraft({ requestId, templateId, variables })
    - On success:
      - Set success message: "Đã tạo bản nháp thành công"
      - Reset form or keep for next generation

    ```tsx
    <Card className="space-y-4">

      <select onChange={handleTemplateChange}>
        <option value="">-- Chọn mẫu tài liệu --</option>
        {templates.map(t => (
          <option key={t.id} value={t.id}>{t.label} (v{t.version})</option>
        ))}
      </select>

      {selectedTemplate && (
        <form onSubmit={handleSubmit} className="space-y-4">
          {selectedTemplate.variableSchema.map(variable => (

                {variable.label}

          {error && <p className="text-red-600">{error}</p>}
          {success && <p className="text-green-600">{success}</p>}
    ```
