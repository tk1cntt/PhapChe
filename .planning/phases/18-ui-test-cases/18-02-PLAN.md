```typescript
import { test, expect } from '@playwright/test';
import { loginAs } from './helpers';

test.describe('Specialist Queue', () => {
  test.beforeEach(async ({ page }) => {
    await loginAs(page, 'specialist');
  });

  test('renders specialist queue page', async ({ page }) => {
    await page.goto('/specialist/requests');
    await expect(page.locator('text=Hàng chờ xử lý')).toBeVisible();
  });

  test('shows queue list', async ({ page }) => {
    await page.goto('/specialist/requests');
    // Wait for content to load
    await page.waitForTimeout(2000);
    // Check for queue items or empty state
    const hasContent = await page.locator('.ant-list-item, text=Chưa có yêu cầu').count() > 0;
    expect(hasContent).toBeTruthy();
  });

  test('can click on request item', async ({ page }) => {
    await page.goto('/specialist/requests');
    await page.waitForTimeout(2000);
    const items = page.locator('.ant-list-item');
    const count = await items.count();
    if (count > 0) {
      await items.first().click();
      await page.waitForTimeout(2000);
      // Should navigate or show detail
      const url = page.url();
      expect(url.includes('/specialist/requests/')).toBeTruthy();
    }
  });

  test('shows triage status', async ({ page }) => {
    await page.goto('/specialist/requests');
    await page.waitForTimeout(1000);
    // Check for triage-related content
    const hasTriage = await page.getByText('triage').count() > 0 || 
                      await page.getByText('Cần phân loại').count() > 0;
    expect(typeof hasTriage).toBe('boolean');
  });
});
```
