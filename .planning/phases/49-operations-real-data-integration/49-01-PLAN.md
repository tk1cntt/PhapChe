   ```typescript
     level: 'ok' | 'warn' | 'danger' | 'info';
     label: string;
     percent: number;
     source: 'deadline' | 'status_age' | 'request_age' | 'none';
   };
   ```

2. Add `OpsAggregateDto` type combining all screen data:
   ```typescript
   export type OpsAggregateDto = {
     stats: {
       openRequests: number;
       nearSla: number;
       completedToday: number;
       auditWarnings: number;
     };
     workload: OpsWorkloadRowDto[];
     timeline: OpsTimelineItemDto[];
     requests: OpsRequestRowDto[];
     filters: {
       workspaces: Array<{ id: string; name: string }>;
       matterTypes: Array<{ key: string; label: string }>;
       specialists: Array<{ id: string; name: string }>;
       reviewers: Array<{ id: string; name: string }>;
       statuses: string[];
     };
     pagination?: { page: number; pageSize: number; total: number };
   };
   ```

3. Add `calcOpsSla(request, latestTransition)` function:
   - If `request.slaDeadline` exists: calculate hours remaining, return `ok` (green, >72h), `warn` (orange, 24-72h), `danger` (red, <24h or past)
   - Else if `latestTransition.createdAt` exists: derive aging, return `ok/warn/danger/info` based on days
   - Else: return `info` level with `label: 'Chưa có SLA'`, `percent: 0`, `source: 'none'`
   - Percent calculation: `(daysElapsed / maxDays) * 100` clamped to 0-100

4. Add `getOpsAggregate(session, filters, pagination)` function:
   - Reuse `requireSessionWorkspace` pattern from `getOpsDashboard`
   - Query Prisma for stats: open requests count, near-SLA count (where deadline within 24h), completed today count, audit warning count
   - Reuse existing workload query from `getOpsDashboard`
   - Query global timeline: merge `AuditEvent` and `WorkflowTransition` ordered by `createdAt desc`, take latest 20
   - Query requests with SLA data, apply filters, add pagination metadata
   - Return typed `OpsAggregateDto`

5. Update `OpsRequestRowDto` to include:
   ```typescript
   slaDeadline: Date | null;
   sla: OpsSlaDto;
   code: string | null;  // per D-04 open question: use code when present
   ```

6. Add filter support for search (title/code contains), matterType, date range.

**DO NOT** hardcode any sample values. All data must come from Prisma queries.
