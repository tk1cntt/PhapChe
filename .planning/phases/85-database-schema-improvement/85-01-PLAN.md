    -- LegalRequest.status CHECK
    CREATE INDEX idx_legal_request_status ON LegalRequest(status);

    -- LegalRequest.priority CHECK (allowed values: low, medium, high, urgent)
    CREATE INDEX idx_legal_request_priority ON LegalRequest(priority);

    -- WorkspaceMembership.role CHECK (allowed values: customer, specialist, reviewer, coordinator, admin)
    CREATE INDEX idx_workspace_membership_role ON WorkspaceMembership(role);

    -- RequestAssignment.kind CHECK (allowed values: specialist, reviewer)
    CREATE INDEX idx_request_assignment_kind ON RequestAssignment(kind);

    -- DocumentVersion.status CHECK (allowed values: draft, submitted_for_review, final)
    CREATE INDEX idx_document_version_status ON DocumentVersion(status);

    -- Review.status CHECK (allowed values: in_progress, approved, rejected)
    CREATE INDEX idx_review_status ON Review(status);

    -- Review.decision CHECK (allowed values: approve, reject)
    CREATE INDEX idx_review_decision ON Review(decision);
