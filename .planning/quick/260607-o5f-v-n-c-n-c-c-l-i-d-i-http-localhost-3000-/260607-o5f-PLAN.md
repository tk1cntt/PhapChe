2. The customRequest should:
   - Get requestId from parent form or component state
   - Call attachIntakeFileAction with FormData containing 'file' and 'requestId'
   - On success, call onSuccess callback and optionally refresh file list
   - On error, call onError callback

Use pattern:
```typescript
const uploadProps = {
  accept: ".pdf,.doc,.docx,.png,.jpg,.jpeg",
  showUploadList: false,
  name: "file",
  customRequest: async (options) => {
    const { file, onSuccess, onError } = options;
    const formData = new FormData();
    formData.append('file', file as File);
    formData.append('requestId', requestId);
    try {
      const result = await attachIntakeFileAction(formData);
      onSuccess(result);
    } catch (error) {
      onError(error as Error);
    }
  }
};
```
